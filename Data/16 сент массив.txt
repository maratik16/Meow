#include <stdio.h>
using namespace std;
int main() {
    int i,j, A[10][10];
    for(i=1; j<=9; i++)
    {
        for(j=1; j<=9; j++)
        {
            A[i][j]=i*j;
            printf("%5d",A[i][j]);
        }
        printf("\n");
    }
    return 0;
}
